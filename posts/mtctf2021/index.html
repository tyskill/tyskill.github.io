import requests
import time
import string
import re

def ord2hex(letter):
    result = ''
    for i in letter:
        result += hex(ord(i))
    result = result.replace('0x','')
    return '0x'+result

def hex2ord(letter):
    result = ''
    for i in range(0,len(letter),2):
        result += chr(int(letter[i:i+2], 16))
    return result
